# [back] package ts-node, sobe o server express
    npx ts-node --ignore-watch node-modules src/server.ts

# [back] package ts-node-dev, sobe o server
    npx ts-node-dev --ignore-watch node-modules  src/server.ts

# [front] ferramenta somente executada que cria um projeto React completo
    npx create-react-app web --template=typescript

# [database] Knex.js, biblioteca de integração com base de dados
    npm install knex
    npm install sqlite3

    # utilizando migrations
        npx knex migrate:latest --knexfile knexfile.ts migrate:latest

    # instalar extensão do VsCode SQLite para visualizar o database criado
        [Windows] + [Ctrl] + [P] : SQLite -> Open Database
        Ira abrir o SQLite explorer na barra lateral com o resultado da base
        
        Para executar uma query, botão direito na base, new query
        [Windows] + [Ctrl] + [P] : SQLite -> Run Query
    
    # popular base de dados com valores default, utilizamos o recurso seed do Knex
        knex --knexfile knexfile.ts seed:run
